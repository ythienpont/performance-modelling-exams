% Define Problem and Solution environments
\newcounter{problemcounter} % Counter for problems
\newenvironment{problem}[1]{
    \stepcounter{problemcounter} % Increment the problem counter
    \addcontentsline{toc}{subsection}{\theproblemcounter. #1} % Add to ToC
    \noindent \textsc{\theproblemcounter. #1:} % Problem title
    \noindent
}{
    \par
    \vspace{1em} % Adjust the space as needed
}

\newenvironment{solution}{
    \noindent \textbf{Solution:}
}{
    \par
    \vspace{1em} % Adjust the space as needed
}

\newcommand{\currentexamyear}{}
% Command to reset the problem counter and create exam title
\newcommand{\exam}[1]{
    \newpage % Start each exam on a new page
    \setcounter{problemcounter}{0} % Reset problem counter
    \renewcommand{\currentexamyear}{#1} % Store the current exam year as a variable
    \noindent \Large \textbf{\uppercase{OPEN BOOK EXAM \currentexamyear}}
    \normalsize \\% Larger exam title
    \thispagestyle{empty} % No header/footer on this page
    \addcontentsline{toc}{section}{Open Book Exam \currentexamyear} % Add to ToC
    \par
}
